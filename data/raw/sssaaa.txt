FORM load_files .

  DATA : wlt_files TYPE TABLE OF epsfili,
         wls_file  LIKE LINE OF wlt_files.

  DATA : wlv_exist  TYPE flag,
         wlv_traite TYPE flag.

  DATA :  wlv_mask TYPE epsf-epsfilnam.

  wlv_mask = |MC_*|.

  CALL FUNCTION 'EPS_GET_DIRECTORY_LISTING'
    EXPORTING
      dir_name               = p_dir
      file_mask              = wlv_mask
    TABLES
      dir_list               = wlt_files
    EXCEPTIONS
      invalid_eps_subdir     = 1
      sapgparam_failed       = 2
      build_directory_failed = 3
      no_authorization       = 4
      read_directory_failed  = 5
      too_many_read_errors   = 6
      empty_directory_list   = 7
      OTHERS                 = 8.

  SORT wlt_files.

  LOOP AT wlt_files INTO wls_file.

    CLEAR : wlv_exist, wlv_traite.

    PERFORM chech_file USING wls_file-name CHANGING wlv_exist wlv_traite.

    IF wlv_exist IS INITIAL.
*      TODO Message nouveau fichier intégré et a traité
      PERFORM read_file USING wls_file-name ''.
      PERFORM move_file_ok USING wls_file-name ''.
    ELSE.
      IF wlv_traite IS NOT INITIAL.
*       TODO Message fichier déjà intégré et traité
      ELSE.
*        TODO Message fichier déjà intégré et a traité
        SELECT * FROM zplm_files_load APPENDING TABLE wgt_extract WHERE file_name_tech = wls_file-name.
      ENDIF.
    ENDIF.

  ENDLOOP.

ENDFORM.

FORM read_file  USING    wiv_file TYPE epsfilnam
                         wiv_sp TYPE flag.

  DATA : wlv_line  TYPE string,
         wlt_line  TYPE TABLE OF string,
         wlv_zone  TYPE string,
         wlv_tabix TYPE sy-tabix.

  DATA : wlv_chemin TYPE string,
         wlv_test   TYPE string.

  DATA : wls_extract   LIKE LINE OF wgt_extract,
         wls_extract2  LIKE LINE OF wgt_extract2,
         wlv_timestamp TYPE timst_kk.

  DATA : wlv_atinn TYPE cawnt-atinn.

  DATA : wlv_num_ligne TYPE int2.

  DATA : wlv_nb_ligne TYPE int4.

  FIELD-SYMBOLS : <wlf_extract_field> TYPE any.

  wlv_timestamp = |{ sy-datum }{ sy-uzeit }|.

  IF wiv_sp IS INITIAL.
    wlv_test = p_dir.
  ELSE.
    wlv_test = p_dir2.
  ENDIF.

  FIND REGEX '/$' IN wlv_test.
  IF sy-subrc <> 0.
    IF wiv_sp IS INITIAL.
      wlv_chemin = |{ p_dir }/{ wiv_file }|.
    ELSE.
      wlv_chemin = |{ p_dir2 }/{ wiv_file }|.
    ENDIF.
  ELSE.
    IF wiv_sp IS INITIAL.
      wlv_chemin = |{ p_dir }{ wiv_file }|.
    ELSE.
      wlv_chemin = |{ p_dir2 }{ wiv_file }|.
    ENDIF.
  ENDIF.


  OPEN DATASET wlv_chemin IN TEXT MODE FOR INPUT ENCODING NON-UNICODE.
  IF sy-subrc = 0.

    DO.
      READ DATASET wlv_chemin INTO wlv_line.
      IF sy-subrc = 0.

        IF p_entete IS NOT INITIAL AND sy-index = 1.
          CONTINUE.
        ENDIF.

        ADD 1 TO wlv_num_ligne.

        REFRESH wlt_line.
        CLEAR : wlv_tabix.
        SPLIT wlv_line AT wgc_separator INTO TABLE wlt_line.
        DESCRIBE TABLE wlt_line LINES wlv_nb_ligne.
        IF wlv_nb_ligne = 1.
          CONTINUE.
        ENDIF.

        CLEAR wls_extract.

        LOOP AT wlt_line INTO wlv_zone.

          IF sy-tabix = 1.
            wlv_tabix = sy-tabix + 1.
          ELSE.
            wlv_tabix = sy-tabix + 5.
          ENDIF.

          IF wiv_sp IS INITIAL.
            ASSIGN COMPONENT wlv_tabix OF STRUCTURE wls_extract TO <wlf_extract_field>.
          ELSE.
            ASSIGN COMPONENT wlv_tabix OF STRUCTURE wls_extract2 TO <wlf_extract_field>.
          ENDIF.

          IF sy-tabix = 21.
            TRANSLATE wlv_zone USING ',.'.
          ENDIF.

          IF wlv_zone = cl_abap_char_utilities=>cr_lf(1) OR
             wlv_zone = cl_abap_char_utilities=>cr_lf+1(1) OR
            wlv_zone = cl_abap_char_utilities=>cr_lf.
            CLEAR wlv_zone.
          ENDIF.

          REPLACE ALL OCCURRENCES OF cl_abap_char_utilities=>cr_lf(1) IN wlv_zone WITH ''.
          REPLACE ALL OCCURRENCES OF cl_abap_char_utilities=>cr_lf+1(1) IN wlv_zone WITH ''.
          REPLACE ALL OCCURRENCES OF cl_abap_char_utilities=>cr_lf IN wlv_zone WITH ''.

          IF wlv_zone = 'x'.
            wlv_zone = 'X'.
          ENDIF.

* BEGIN INSERT RZE 29/06/2023 RICEF-100
* Gestion des tailles min et max en lettre envoyées par PLM
          IF wlv_tabix = '84' OR wlv_tabix = '85' OR wlv_tabix = '89' OR wlv_tabix = '90'.
            CALL FUNCTION 'CONVERSION_EXIT_ATINN_INPUT'
              EXPORTING
                input  = wls_extract-grillet
              IMPORTING
                output = wlv_atinn.

* Taille min
            SELECT SINGLE atzhl FROM cawnt
                                INTO @DATA(wlv_atzhl)
                                WHERE atinn = @wlv_atinn
                                AND   spras = @sy-langu
                                AND   atwtb = @wlv_zone.
            IF sy-subrc = 0.
              wlv_zone = wlv_atzhl.
            ENDIF.

          ENDIF.
* END INSERT RZE 29/06/2023 RICEF-100


* DEB INS ABAP-2071 HBO
          IF wlv_tabix = '119'. "sy-tabix 115
            wls_extract-nompers = wlv_zone.
          ENDIF.
* END INS ABAP-2071 HBO
          IF <wlf_extract_field> IS ASSIGNED."ABAP-2071 HBO
            <wlf_extract_field> = wlv_zone.
          ENDIF."ABAP-2071 HBO
          UNASSIGN  <wlf_extract_field>."ABAP-2071 HBO
*          <wlf_extract_field> = wlv_zone." -- ABAP-2071 HBO

        ENDLOOP.

* DEB INS ABAP-2593 MSA
        " Nettoyage du module linéaire
        IF wls_extract-laygr IS NOT INITIAL.
          " Supprimer les guillemets
          REPLACE ALL OCCURRENCES OF '"' IN wls_extract-laygr WITH ''.
          " Supprimer les espaces en début/fin
          CONDENSE wls_extract-laygr.
        ENDIF.

* END INS ABAP-2593 MSA

        IF wiv_sp IS INITIAL.
          wls_extract-mandt = sy-mandt.
          wls_extract-file_name_tech = wiv_file.
          wls_extract-timestamp_tech = wlv_timestamp.
          wls_extract-statut_tech = ''.
          wls_extract-num_ligne = wlv_num_ligne.
          APPEND wls_extract TO wgt_extract.
        ELSE.
          wls_extract2-mandt = sy-mandt.
          wls_extract2-file_name_tech = wiv_file.
          wls_extract2-timestamp_tech = wlv_timestamp.
          wls_extract2-statut_tech = ''.
          wls_extract2-num_ligne = wlv_num_ligne.
          APPEND wls_extract2 TO wgt_extract2.
        ENDIF.

      ELSE.
        EXIT.
      ENDIF.

    ENDDO.

    CLOSE DATASET wlv_chemin.

  ENDIF.

ENDFORM.

FORM chech_file  USING    wiv_file TYPE epsfilnam
                 CHANGING wev_exist TYPE flag
                          wev_traite TYPE flag.

  DATA : wls_file_load TYPE zplm_files_load.

*gestion des statuts de la table
* '' = intégré dans la table
* 0 = intégré et check effectué avec succés
* 1 = intégré et traité
* 2 = intégré et check en erreur
*
  SELECT SINGLE * FROM zplm_files_load INTO wls_file_load WHERE file_name_tech = wiv_file AND ekorg = p_ekorg.
  IF sy-subrc = 0.
    wev_exist = 'X'.
    IF wls_file_load-statut_tech = 1.
    ELSE.
      SELECT SINGLE * FROM zplm_files_load INTO wls_file_load WHERE file_name_tech = wiv_file AND statut_tech = 1 AND ekorg = p_ekorg.
      IF sy-subrc = 0.
        wev_traite = 'X'.
      ENDIF.
    ENDIF.

  ENDIF.
